# ==========================================
# DAY 3 : TERMINAL BASICS
# ==========================================

# OBJECTIVE:
# Linux terminal ke basic navigation commands seekhna


# ------------------------------------------
# pwd = current location dikhata hai
# ------------------------------------------

# use:
# Main abhi kis folder me hu

# example:
pwd

# output:
/home/kali/Desktop


# ------------------------------------------
# ls = current folder ke files/folders dikhata hai
# ------------------------------------------

# use:
# Folder ke andar kya kya hai dekhna

# example:
ls

# output:
file1.txt  folder1


# ------------------------------------------
# ls -l = detail view
# ------------------------------------------

# use:
# Size, permission, date details dekhna

# example:
ls -l


# ------------------------------------------
# ls -a = hidden files bhi dikhata hai
# ------------------------------------------

# hidden files start with "."

# examples:
.bashrc
.config
.cache

# command:
ls -a


# ------------------------------------------
# ls -la = hidden + detail dono
# ------------------------------------------

# example:
ls -la


# ------------------------------------------
# cd foldername = folder ke andar jana
# ------------------------------------------

# use:
# Location change karna

# example:
cd Desktop


# ------------------------------------------
# cd .. = ek step back jana
# ------------------------------------------

# use:
# Parent folder me jana

# example:
cd ..


# ------------------------------------------
# cd ~ = home directory me jana
# ------------------------------------------

# example:
cd ~


# ------------------------------------------
# cd / = root directory me jana
# ------------------------------------------

# example:
cd /


# ------------------------------------------
# cd = same as cd ~
# ------------------------------------------

# example:
cd


# ------------------------------------------
# clear = terminal clean karna
# ------------------------------------------

# example:
clear

# shortcut:
Ctrl + L


# ------------------------------------------
# IMPORTANT CONCEPTS
# ------------------------------------------

# .. = ek level upar
# . = current folder

# example:
ls .


# ------------------------------------------
# COMMON WORKFLOW
# ------------------------------------------

# example:
pwd
ls
cd Desktop
ls
