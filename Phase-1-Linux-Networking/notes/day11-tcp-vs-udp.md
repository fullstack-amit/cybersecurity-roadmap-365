# ------------------------------------------

# TCP = reliable, slow

# features:
# - connection based
# - error check
# - resend

# use:
# login
# file transfer
# banking


# ------------------------------------------
# TCP FLOW
# ------------------------------------------

# connect
# send
# confirm
# resend if error


# ------------------------------------------
# UDP
# ------------------------------------------

# UDP = fast, no guarantee

# features:
# - no connection
# - no error check
# - fast


# ------------------------------------------
# UDP FLOW
# ------------------------------------------

# send
# done


# ------------------------------------------
# UDP USES
# ------------------------------------------

# video streaming
# online gaming
# voice calls


# ------------------------------------------
# QUICK DIFFERENCE
# ------------------------------------------

# TCP = accuracy
# UDP = speed

# fast but unreliable
